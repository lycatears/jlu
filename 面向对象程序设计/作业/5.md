//        Time.h

#include <iostream>
using namespace std;
class Time
{ // 时间类定义
private:
    int hour;   // 时
    int minute; // 分
    int second; // 秒
public:
    Time(int hh = 0, int mm = 0, int ss = 0); // 指定参数的构造函数
    ~Time();                                  // 析构函数
    void ShowTime();                          // 显示时间类中存储的时间
    void SetTime();                           // 通过用户输入设置时间
    void SetTime(int hh, int mm, int ss);     // 通过调用函数修改时间
    operator int();
    friend Time operator+(Time &a, Time &b);//两个时间对象相加
    friend Time operator+(Time &a, int ss);//时间对象与整数相加
    friend Time operator-(Time &a, Time &b);//两个时间对象相减
    friend Time operator-(Time &a, int ss);//时间对象与整数相减
};
//-----------------------------
//        Time.cpp
//-----------------------------
#include "Time.h"
Time::Time(int hh, int mm, int ss)
{
    // 指定参数的构造函数，默认值为全0
    // 输入：时分秒
    // 输出：新的时间对象
    if (hh < 0 || mm < 0 || ss < 0)
    {
        cout << "Failed to construct a new Time!" << endl;
        hh = mm = ss = 0;
    }
    if (ss >= 60)
    {
        mm += ss / 60;
        ss %= 60;
    }
    if (mm >= 60)
    {
        hh += mm / 60;
        mm %= 60;
        hh %= 24;
    }
    hour = hh;
    minute = mm;
    second = ss;
    return;
}
Time::~Time()
{ // 析构函数
}
void Time::ShowTime()
{
    // 显示时间对象的时间信息
    // 输入：无
    // 输出：无
    printf("%02d:%02d:%02d\n", hour, minute, second);
    return;
}
void Time::SetTime()
{
    // 通过用户输入设置事件对象的时间信息
    // 输入：无
    // 输出：无
    cout << "Enter New Time";
    while (1)
    {
        cin >> hour >> minute >> second;
        if (cin.rdstate() != 0)
        {//输入非法处理
            cout << "Invalid time data.Please input again:";
            cin.clear();
            cin.sync();
            continue;
        }
        if (hour < 24 && hour >= 0 && minute < 60 && minute >= 0 && second < 60 && second >= 0 || cin.rdstate() == 0)
        { // 检查数据是否合法
            break;
        }
        cout << "Invalid time data.Please input again:";
    }
    return;
}
void Time::SetTime(int hh, int mm, int ss)
{
    // 通过调用函数设置时间对象的时间信息
    // 输入：无
    // 输出：无
    if (ss >= 60)
    {
        mm += ss / 60;
        ss %= 60;
    }
    if (mm >= 60)
    {
        hh += mm / 60;
        mm %= 60;
        hh %= 24;
    }
    hour = hh;
    minute = mm;
    second = ss;
    return;
}
Time operator+(Time &a, Time &b)
{
    // 运算符重载，便于时间对象相加
    Time c;
    c.hour = a.hour + b.hour;
    c.minute = a.minute + b.minute;
    c.second = a.second + b.second;
    if (c.second >= 60)
    {
        c.minute++;
        c.second %= 60;
    }
    if (c.minute >= 60)
    {
        c.hour++;
        c.minute %= 60;
    }
    if (c.hour >= 24)
    {
        c.hour %= 24;
    }
    return c;
}
Time operator+(Time &a, int ss)
{
    ss += (int)a;
    while (ss < 0)
    {
        ss += 86400;
    }
    return Time(0, 0, ss);
}
Time::operator int()
{ // 类型转换运算符重载，将Time类转换为int型秒数
    return hour * 3600 + minute * 60 + second;
}
Time operator-(Time &a, Time &b)
{
    return Time(0, 0, abs((int)a - (int)b));
}
Time operator-(Time &a, int ss)
{
    ss = (int)a - ss;
    while (ss < 0)
    {
        ss += 86400;
    }
    return Time(0, 0, ss);
}
